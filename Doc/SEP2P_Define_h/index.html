//SEP2P_MSG_GET_ALARM_INFO_RESP 
typedef struct tag_MSG_GET_ALARM_INFO_RESP { 
    INT32 bMDEnable; //0->disable; 1->enable 
    INT32 nMDSensitivity; //[0,9],0->the highest, 9->the lowest 
    INT32 bInputAlarm; //0->disable; 1->enable 
    INT32 nInputAlarmMode;//0->low level; 1->high level 
    INT32 bIOLinkageWhenAlarm; //0->disable linkage; 1->enable linkage 
    INT32 reserve1; 
    INT32 nPresetbitWhenAlarm; //[0,8]; 0->disable; >0 enable, position index from 1 to 8 
    INT32 bMailWhenAlarm; //0->disable; 1->enable 
    INT32 bSnapshotToSDWhenAlarm; //0->disable; 1->enable 
    INT32 bRecordToSDWhenAlarm; //0->disable; 1->enable 
    INT32 bSnapshotToFTPWhenAlarm; //0->disable; 1->enable 
    INT32 bRecordToFTPWhenAlarm; //0->disable; 1->enable 
    CHAR reserve2[92]; 
}MSG_GET_ALARM_INFO_RESP, *PMSG_GET_ALARM_INFO_RESP;
Fields
bMDEnable

type: INT32 ; means: alarm type .

nMDSensitivity

type: INT32 ; means: motion detect alarm sensitivity. its level is from 0 to 9. the highest is 0, the lowest is 9.

bInputAlarm

type: INT32 ; means: whether to enable input alarm, 0 is disable, 1 is enable.

nInputAlarmMode

type: INT32 ; means:input alarm mode. 0 is low level mode, 1 is high level mode.

bIOLinkageWhenAlarm

type: INT32 ; means:whether to enable io linkage when alarm, 0 is disable, 1 is enable.

reserve1

type: INT32 ; means:no used.

nPresetbitWhenAlarm

type: INT32 ; means: set preset position index whan alarm. Disable it if the value is 0. Enable it if the value is more than 0. The range of position index is from 1 to 8.

bMailWhenAlarm

type: INT32 ; means: whether to send E-Mail when alarm. 0 is disable, 1 is enable.The FTP setting refer to SEP2P_MSG_SET_EMAIL_INFO_REQ.

bSnapshotToSDWhenAlarm

type: INT32 ; means: whether to snapshot to SDCard when alarm. 0 is disable, 1 is enable.

bRecordToSDWhenAlarm

type: INT32 ; means: whether to record to SDCard when alarm. 0 is disable, 1 is enable.

bSnapshotToFTPWhenAlarm

type: INT32 ; means: whether to snapshot to FTP server when alarm. 0 is disable, 1 is enable.The FTP setting refer to SEP2P_MSG_SET_FTP_INFO_REQ.

bRecordToFTPWhenAlarm

type: INT32 ; means: whether to record to FTP server when alarm. 0 is disable, 1 is enable.The FTP setting refer to SEP2P_MSG_SET_FTP_INFO_REQ.

reserve2
